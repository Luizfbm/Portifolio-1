História

O IAeJovem foi pensado como uma ponte entre acolhimento e gestão escolar. O aluno precisa de um espaço confidencial para falar; a instituição precisa identificar padrões de risco sem invadir essa privacidade. A solução separa conversa privada, score emocional e acompanhamento pedagógico, criando um produto sensível ao contexto humano e tecnicamente estruturado para crescer.

function analyzeEmotions(messages: ConversationMessage[]): { score: number, emocao: string } {
  const emotionWords = {
    feliz: ["bem", "bom", "ótimo", "feliz", "alegre"],
    triste: ["triste", "mal", "ruim", "deprimido", "sozinho"],
    ansioso: ["ansioso", "preocupado", "estressado", "nervoso"],
  };

  let positiveCount = 0;
  let negativeCount = 0;

  userMessages.forEach((message) => {
    const words = message.content.toLowerCase().split(/\s+/);
    words.forEach((word) => {
      Object.entries(emotionWords).forEach(([emotion, keywords]) => {
        if (keywords.some((keyword) => word.includes(keyword))) {
          if (emotion === "feliz" || emotion === "tranquilo") positiveCount++;
          else negativeCount++;
        }
      });
    });
  });

  const totalEmotionalWords = positiveCount + negativeCount;
  const sentimentoScore = totalEmotionalWords > 0
    ? (positiveCount - negativeCount) / totalEmotionalWords
    : 0;
  const baseScore = 5 + (sentimentoScore * 5);

  return {
    score: Math.round(Math.max(0, Math.min(10, baseScore)) * 10) / 10,
    emocao: "determinado_pelo_algoritmo"
  };
}
